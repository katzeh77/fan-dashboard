const client = mqtt.connect("wss://broker.emqx.io:8084/mqtt"); let schedule = Array(24).fill().map(()=>Array(8).fill(0)); let selectedChannel = 0; let selectedHour = 0; // warna channel const colors = [ "#0033ff", // royal blue "#00ccff", // blue "#9900ff", // violet "#ff0000", // red "#00ff00", // green "#ffffff", // white "#cc00ff", // sump "#ffaa00" // fan ]; // ================= MQTT ================= client.on("connect",()=>{ console.log("MQTT Connected"); }); function send(){ client.publish("reef/schedule", JSON.stringify({data:schedule})); } // ================= CHART ================= const ctx = document.getElementById("chart"); const chart = new Chart(ctx, { type: 'line', data: { labels: [...Array(24).keys()], datasets: colors.map((c,i)=>({ label: "CH"+(i+1), borderColor: c, data: schedule.map(h=>h[i]), tension: 0.3, pointRadius: 5 })) }, options: { responsive: true, interaction: { mode: 'nearest', intersect: true }, scales: { y: { min: 0, max: 100 } }, onClick: (evt, elements) => { if(elements.length > 0){ const i = elements[0].index; const d = elements[0].datasetIndex; selectedHour = i; selectedChannel = d; console.log("Edit Hour:", i, "Channel:", d); } } } }); // ================= UPDATE GRAPH ================= function redraw(){ chart.data.datasets.forEach((ds,i)=>{ ds.data = schedule.map(h=>h[i]); }); chart.update(); } // ================= DRAG CHANGE ================= ctx.addEventListener("mousedown", startDraw); ctx.addEventListener("mousemove", drawMove); ctx.addEventListener("mouseup", stopDraw); let drawing = false; function startDraw(e){ drawing = true; updateValue(e); } function drawMove(e){ if(!drawing) return; updateValue(e); } function stopDraw(){ drawing = false; } function updateValue(e){ const rect = ctx.getBoundingClientRect(); const y = e.clientY - rect.top; let value = 100 - (y / rect.height * 100); value = Math.max(0, Math.min(100, value)); schedule[selectedHour][selectedChannel] = value; redraw(); send(); } // ================= PRESET ================= function savePreset(name){ localStorage.setItem(name, JSON.stringify(schedule)); } function loadPreset(name){ let d = localStorage.getItem(name); if(d){ schedule = JSON.parse(d); redraw(); send(); } } // ================= TEMP ================= function setTemp(){ let t = document.getElementById("maxTemp").value; client.publish("reef/maxtemp", JSON.stringify({maxTemp:parseFloat(t)})); }
