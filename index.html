let drawing = false; // ================= DESKTOP ================= ctx.addEventListener("mousedown", e=>{ drawing = true; updateValue(e.clientX, e.clientY); }); ctx.addEventListener("mousemove", e=>{ if(!drawing) return; updateValue(e.clientX, e.clientY); }); ctx.addEventListener("mouseup", ()=>{ drawing = false; }); // ================= HP / TOUCH ================= ctx.addEventListener("touchstart", e=>{ e.preventDefault(); drawing = true; let touch = e.touches[0]; updateValue(touch.clientX, touch.clientY); }); ctx.addEventListener("touchmove", e=>{ e.preventDefault(); if(!drawing) return; let touch = e.touches[0]; updateValue(touch.clientX, touch.clientY); }); ctx.addEventListener("touchend", ()=>{ drawing = false; });
